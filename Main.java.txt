import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Library library = new Library();
        Scanner sc = new Scanner(System.in);

        while (true) {
            System.out.println("\n--- Library Menu ---");
            System.out.println("1. Add Book");
            System.out.println("2. Register User");
            System.out.println("3. Borrow Book");
            System.out.println("4. Return Book");
            System.out.println("5. View Available Books");
            System.out.println("6. View Borrow History");
            System.out.println("7. Exit");
            System.out.print("Choose an option: ");

            int choice = sc.nextInt();
            sc.nextLine(); // consume newline

            switch (choice) {
                case 1:
                    System.out.print("Book ID: "); String bid = sc.nextLine();
                    System.out.print("Title: "); String title = sc.nextLine();
                    System.out.print("Author: "); String author = sc.nextLine();
                    library.addBook(new Book(bid, title, author, true));
                    break;
                case 2:
                    System.out.print("User ID: "); String uid = sc.nextLine();
                    System.out.print("Name: "); String name = sc.nextLine();
                    library.registerUser(new User(uid, name));
                    break;
                case 3:
                    System.out.print("User ID: "); String borrowUid = sc.nextLine();
                    System.out.print("Book ID: "); String borrowBid = sc.nextLine();
                    library.borrowBook(borrowUid, borrowBid);
                    break;
                case 4:
                    System.out.print("User ID: "); String returnUid = sc.nextLine();
                    System.out.print("Book ID: "); String returnBid = sc.nextLine();
                    library.returnBook(returnUid, returnBid);
                    break;
                case 5:
                    library.viewAvailableBooks();
                    break;
                case 6:
                    System.out.print("User ID: "); String historyUid = sc.nextLine();
                    library.viewBorrowHistory(historyUid);
                    break;
                case 7:
                    System.out.println("Exiting...");
                    sc.close();
                    return;
                default:
                    System.out.println("Invalid option. Try again.");
            }
        }
    }
}
